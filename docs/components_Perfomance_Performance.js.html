import React from 'react';
import {Legend, PolarAngleAxis, PolarGrid, Radar, RadarChart, ResponsiveContainer} from "recharts";
import capitalizeFirstLetter from "../../sevices/Helpers/capitalizeFirstLetter";
import './Performance.css'

/**
 * @typedef {Object} PerformanceDataItem
 * @property {string} kind - Тип производительности.
 * @property {number} value - Значение производительности.
 */

/**
 * Компонент Performance отображает данные производительности в виде радарного графика.
 *
 * @component
 * @param {Object} props - Свойства компонента.
 * @param {Object} props.performance - Объект производительности, содержащий массив данных о производительности.
 * @param {PerformanceDataItem[]} props.performance.data - Массив элементов данных производительности.
 * @returns {JSX.Element}
 */
const Performance = ({performance}) => {
    const transformPerformance = performance.data.map(el => ({...el, subject: capitalizeFirstLetter(performance.kind[el.kind])}))
    return (
        <div className="performance">
            <ResponsiveContainer width='100%' height='100%'>
                <RadarChart outerRadius={70} width='100%' height='100%' data={transformPerformance} >
                    <PolarGrid polarRadius={[10,20,40,55]} strokeLinecap={'butt'}  radialLines={false}/>
                    <PolarAngleAxis dataKey="subject" stroke='#FFFFFF' tickLine={false}
                                    tick={{fontSize: '12px'}}/>
                    <Radar dataKey="value" fill="#FF0101B2" fillOpacity={0.95} legendType='none'/>
                    <Legend/>
                </RadarChart>
            </ResponsiveContainer>
        </div>

    );
};

export default Performance;
