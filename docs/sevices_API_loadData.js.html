import axios from "axios";

/**
 * Load data for a user from an API.
 * @async
 * @param {string} id - The ID of the user.
 * @returns {Promise<{user: Object, activity: Object, averageSession: Object, performance: Object}|{message: string}>} An object containing user, activity, averageSession, and performance data, or an object containing an error message.
 */
const loadData = async (id) => {
    try {
        const [user, activity, averageSession, performance] = await Promise.all([
            axios.get(`http://localhost:3000/user/${id}/`),
            axios.get(`http://localhost:3000/user/${id}/activity`),
            axios.get(`http://localhost:3000/user/${id}/average-sessions`),
            axios.get(`http://localhost:3000/user/${id}/performance`)
        ])
        return {
            user: user.data.data,
            activity: activity.data.data,
            averageSession: averageSession.data.data,
            performance: performance.data.data
        }
    } catch (e) {
        return {message: 'Cet utilisateur n\'existe pas'}
    }
}

export default loadData
